    <script>
        // الحصول على عنصر Canvas وسياق الرسم
        const canvas = document.getElementById('radarCanvas');
        const ctx = canvas.getContext('2d');
        
        // ضبط أبعاد Canvas
        canvas.width = canvas.offsetWidth;
        canvas.height = canvas.offsetHeight;
        
        // متغيرات الرادار
        let radarRange = 100; // نطاق الرادار بالكيلومترات
        let scanAngle = 0; // زاوية المسح الحالية
        let scanSpeed = 5; // سرعة المسح (1-10)
        let sensitivity = 7; // حساسية الكشف (1-10)
        let autoMode = false; // الوضع التلقائي
        let targets = []; // قائمة الأهداف
        let detectionLog = []; // سجل الكشف
        let lastTargetId = 3; // آخر معرف مستخدم للهدف
        
        // عناصر التحكم
        const rangeSlider = document.getElementById('rangeSlider');
        const rangeValue = document.getElementById('rangeValue');
        const speedSlider = document.getElementById('speedSlider');
        const speedValue = document.getElementById('speedValue');
        const sensitivitySlider = document.getElementById('sensitivitySlider');
        const sensitivityValue = document.getElementById('sensitivityValue');
        const addTargetBtn = document.getElementById('addTargetBtn');
        const clearTargetsBtn = document.getElementById('clearTargetsBtn');
        const autoModeBtn = document.getElementById('autoModeBtn');
        const resetBtn = document.getElementById('resetBtn');
        const detectionList = document.getElementById('detectionList');
        
        // تهيئة الأهداف العشوائية الأولية
        initializeRandomTargets();
        
        // إعداد مستمعي الأحداث لعناصر التحكم
        rangeSlider.addEventListener('input', function() {
            radarRange = parseInt(this.value);
            rangeValue.textContent = radarRange + ' كم';
        });
        
        speedSlider.addEventListener('input', function() {
            scanSpeed = parseInt(this.value);
            speedValue.textContent = scanSpeed;
        });
        
        sensitivitySlider.addEventListener('input', function() {
            sensitivity = parseInt(this.value);
            sensitivityValue.textContent = sensitivity;
        });
        
        addTargetBtn.addEventListener('click', addRandomTarget);
        
        clearTargetsBtn.addEventListener('click', function() {
            targets = [];
            updateDetectionLog();
        });
        
        autoModeBtn.addEventListener('click', function() {
            autoMode = !autoMode;
            autoModeBtn.textContent = autoMode ? 'إيقاف الوضع التلقائي' : 'الوضع التلقائي';
            autoModeBtn.style.background = autoMode 
                ? 'linear-gradient(90deg, #ff8000, #ffaa00)' 
                : 'linear-gradient(90deg, #0080ff, #00aaff)';
        });
        
        resetBtn.addEventListener('click', function() {
            radarRange = 100;
            scanSpeed = 5;
            sensitivity = 7;
            autoMode = false;
            
            rangeSlider.value = radarRange;
            rangeValue.textContent = radarRange + ' كم';
            
            speedSlider.value = scanSpeed;
            speedValue.textContent = scanSpeed;
            
            sensitivitySlider.value = sensitivity;
            sensitivityValue.textContent = sensitivity;
            
            autoModeBtn.textContent = 'الوضع التلقائي';
            autoModeBtn.style.background = 'linear-gradient(90deg, #0080ff, #00aaff)';
            
            targets = [];
            initializeRandomTargets();
        });
        
        // إعادة ضبط حجم Canvas عند تغيير حجم النافذة
        window.addEventListener('resize', function() {
            canvas.width = canvas.offsetWidth;
            canvas.height = canvas.offsetHeight;
        });
        
        // تهيئة أهداف عشوائية
        function initializeRandomTargets() {
            targets = [];
            for (let i = 0; i < 3; i++) {
                addRandomTarget();
            }
        }
        
        // إضافة هدف عشوائي جديد
        function addRandomTarget() {
            lastTargetId++;
            const distance = Math.random() * radarRange * 0.8 + radarRange * 0.1;
            const angle = Math.random() * 360;
            const speed = Math.random() * 5 + 1;
            const size = Math.random() * 3 + 1;
            
            targets.push({
                id: lastTargetId,
                distance: distance,
                angle: angle,
                speed: speed,
                size: size,
                type: Math.random() > 0.5 ? 'aircraft' : 'vehicle',
                detected: false
            });
            
            updateDetectionLog();
        }
        
        // تحديث سجل الكشف
        function updateDetectionLog() {
            detectionList.innerHTML = '';
            
            // عرض أحدث 10 عمليات كشف
            const recentDetections = targets
                .filter(target => target.detected)
                .slice(-10);
            
            if (recentDetections.length === 0) {
                detectionList.innerHTML = '<div class="detection-item" style="justify-content: center;">لا توجد أهداف مكتشفة</div>';
                return;
            }
            
            recentDetections.forEach(target => {
                const detectionItem = document.createElement('div');
                detectionItem.className = 'detection-item';
                
                detectionItem.innerHTML = `
                    <span class="detection-id">الهدف #${target.id}</span>
                    <span class="detection-distance">مسافة: ${Math.round(target.distance)} كم</span>
                    <span class="detection-angle">زاوية: ${Math.round(target.angle)}°</span>
                `;
                
                detectionList.appendChild(detectionItem);
            });
        }
        
        // رسم الرادار
        function drawRadar() {
            const centerX = canvas.width / 2;
            const centerY = canvas.height / 2;
            const radius = Math.min(centerX, centerY) * 0.85;
            
            // مسح الخلفية
            ctx.fillStyle = '#050a1a';
            ctx.fillRect(0, 0, canvas.width, canvas.height);
            
            // رسم دوائر نطاق الرادار
            ctx.strokeStyle = 'rgba(0, 255, 234, 0.3)';
            ctx.lineWidth = 1;
            
            for (let i = 1; i <= 5; i++) {
                const circleRadius = radius * (i / 5);
                ctx.beginPath();
                ctx.arc(centerX, centerY, circleRadius, 0, Math.PI * 2);
                ctx.stroke();
                
                // كتابة المسافات
                ctx.fillStyle = 'rgba(0, 255, 234, 0.7)';
                ctx.font = '12px Arial';
                ctx.textAlign = 'center';
                const distance = (radarRange * i / 5).toFixed(0);
                ctx.fillText(distance + ' كم', centerX, centerY - circleRadius - 5);
            }
            
            // رسم خطوط الزوايا
            ctx.strokeStyle = 'rgba(0, 255, 234, 0.2)';
            ctx.lineWidth = 1;
            
            for (let angle = 0; angle < 360; angle += 30) {
                const rad = (angle - 90) * Math.PI / 180;
                const x = centerX + radius * Math.cos(rad);
                const y = centerY + radius * Math.sin(rad);
                
                ctx.beginPath();
                ctx.moveTo(centerX, centerY);
                ctx.lineTo(x, y);
                ctx.stroke();
                
                // كتابة درجات الزوايا
                if (angle % 90 === 0) {
                    const labelRadius = radius + 20;
                    const labelX = centerX + labelRadius * Math.cos(rad);
                    const labelY = centerY + labelRadius * Math.sin(rad);
                    
                    ctx.fillStyle = 'rgba(0, 255, 234, 0.8)';
                    ctx.font = 'bold 14px Arial';
                    ctx.textAlign = 'center';
                    ctx.textBaseline = 'middle';
                    ctx.fillText(angle + '°', labelX, labelY);
                }
            }
            
            // رسم خط المسح (مؤشر الرادار)
            const scanRad = (scanAngle - 90) * Math.PI / 180;
            const scanX = centerX + radius * Math.cos(scanRad);
            const scanY = centerY + radius * Math.sin(scanRad);
            
            // رسم تأثير التدرج للخط
            const gradient = ctx.createLinearGradient(centerX, centerY, scanX, scanY);
            gradient.addColorStop(0, 'rgba(0, 255, 234, 0.8)');
            gradient.addColorStop(1, 'rgba(0, 255, 234, 0.1)');
            
            ctx.strokeStyle = gradient;
            ctx.lineWidth = 2;
            ctx.beginPath();
            ctx.moveTo(centerX, centerY);
            ctx.lineTo(scanX, scanY);
            ctx.stroke();
            
            // رسم قطاع المسح
            ctx.fillStyle = 'rgba(0, 255, 234, 0.05)';
            ctx.beginPath();
            ctx.moveTo(centerX, centerY);
            ctx.arc(centerX, centerY, radius, scanRad - 0.1, scanRad + 0.1);
            ctx.closePath();
            ctx.fill();
            
            // تحديث زاوية المسح
            scanAngle += scanSpeed * 0.5;
            if (scanAngle >= 360) scanAngle = 0;
            
            // رسم الأهداف
            targets.forEach(target => {
                // تحويل المسافة والزاوية إلى إحداثيات
                const targetRad = (target.angle - 90) * Math.PI / 180;
                const targetRadius = (target.distance / radarRange) * radius;
                const targetX = centerX + targetRadius * Math.cos(targetRad);
                const targetY = centerY + targetRadius * Math.sin(targetRad);
                
                // تحديد ما إذا كان الهدف ضمن زاوية المسح
                const angleDiff = Math.abs((target.angle - scanAngle + 180) % 360 - 180);
                const isInScanBeam = angleDiff < sensitivity * 3;
                
                // تحديث حالة الكشف إذا كان الهدف ضمن شعاع المسح
                if (isInScanBeam && targetRadius <= radius && !target.detected) {
                    target.detected = true;
                    updateDetectionLog();
                }
                
                // تحديد لون الهدف بناءً على نوعه وحالة الكشف
                let targetColor;
                if (!target.detected) {
                    targetColor = 'rgba(255, 255, 255, 0.3)';
                } else if (target.type === 'aircraft') {
                    targetColor = 'rgba(255, 100, 100, 0.9)';
                } else {
                    targetColor = 'rgba(100, 255, 100, 0.9)';
                }
                
                // رسم الهدف
                ctx.fillStyle = targetColor;
                ctx.beginPath();
                ctx.arc(targetX, targetY, target.size * 3, 0, Math.PI * 2);
                ctx.fill();
                
                // تأثير النبض للأهداف المكتشفة
                if (target.detected) {
                    ctx.strokeStyle = targetColor;
                    ctx.lineWidth = 1;
                    ctx.beginPath();
                    ctx.arc(targetX, targetY, target.size * 5, 0, Math.PI * 2);
                    ctx.stroke();
                }
                
                // كتابة معرف الهدف
                if (target.detected) {
                    ctx.fillStyle = targetColor;
                    ctx.font = 'bold 12px Arial';
                    ctx.textAlign = 'center';
                    ctx.fillText(`#${target.id}`, targetX, targetY - 15);
                }
                
                // تحريك الهدف في الوضع التلقائي
                if (autoMode) {
                    target.angle += target.speed * 0.1;
                    if (target.angle >= 360) target.angle -= 360;
                    
                    target.distance += (Math.random() - 0.5) * 0.5;
                    if (target.distance < 5) target.distance = 5;
                    if (target.distance > radarRange * 0.9) target.distance = radarRange * 0.9;
                }
            });
            
            // رسم المعلومات الإضافية
            ctx.fillStyle = 'rgba(0, 255, 234, 0.9)';
            ctx.font = 'bold 16px Arial';
            ctx.textAlign = 'left';
            ctx.fillText(`نطاق الرادار: ${radarRange} كم`, 20, 30);
            ctx.fillText(`سرعة المسح: ${scanSpeed}`, 20, 60);
            ctx.fillText(`حساسية: ${sensitivity}/10`, 20, 90);
            ctx.fillText(`الأهداف: ${targets.length}`, 20, 120);
            
            // رسم اتجاه الشمال
            ctx.fillStyle = 'rgba(255, 255, 255, 0.8)';
            ctx.font = 'bold 14px Arial';
            ctx.textAlign = 'center';
            ctx.fillText('شمال ↑', centerX, 30);
            
            // رسم دائرة مركزية
            ctx.fillStyle = 'rgba(0, 255, 234, 0.8)';
            ctx.beginPath();
            ctx.arc(centerX, centerY, 5, 0, Math.PI * 2);
            ctx.fill();
        }
        
        // حلقة الرسوم المتحركة
        function animate() {
            drawRadar();
            requestAnimationFrame(animate);
        }
        
        // بدء الرسم
        animate();
        
        // إضافة هدف عشوائي تلقائي في الوضع التلقائي
        setInterval(function() {
            if (autoMode && Math.random() > 0.95) {
                addRandomTarget();
            }
        }, 3000);
    </script>
